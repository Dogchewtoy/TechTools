SELECT 
  V_SUBNETS_USER.CIDR_MASK,
  V_SUBNETS_USER.LOW_ADDRESS,
  V_SUBNETS_USER.HIGH_ADDRESS,
  V_SUBNETS_USER.BROADCAST_ADDRESS,
  V_SUBNETS_USER.SHORT_SUBNET,
  V_SUBNETS_USER.ADDRESS_COUNT,
  V_SUBNETS_USER.CHILD_COUNT,
  V_SUBNETS_USER.CREATED,                     
  V_SUBNETS_USER.UPDATED
FROM
  V_SUBNETS_USER
WHERE
  V_SUBNETS_USER.TREE_PATH STARTS WITH :TREE_PATH OR
  (CAST(:TREE_PATH AS VARCHAR(230)) IS NULL)
ORDER BY
  V_SUBNETS_USER.IP_VERSION,                            
  V_SUBNETS_USER.SUBNET






SELECT
  A.AUTONOMOUS_NETWORK_ID, 
  C.AUTONOMOUS_NETWORK_NAME,
  A.SHORT_SUBNET,
  A.CIDR_MASK,
  A.ID AS SUBNET_ID,
  B.DESCRIPTIVE_SUBNET_PATH
FROM
  V_SUBNETS_USER A
LEFT JOIN
  P_GET_DESCRIPTIVE_SUBNET_PATH(A.ID) B 
ON 
  1=1
JOIN
  AUTONOMOUS_NETWORKS C
ON 
  A.AUTONOMOUS_NETWORK_ID=C.ID
WHERE 
  A.TREE_PATH STARTS WITH :TREE_PATH OR
  CAST(:TREE_PATH AS VARCHAR(230)) IS NULL
ORDER BY
  A.AUTONOMOUS_NETWORK_ID,
  A.ITEM_TYPE,                                                                           
  A.SUBNET,
  A.CIDR_MASK
